104 # regarding *_PRI_LOGGER_HEADER_COLOR_PREFIX, defined bellow:
105 # HIGH, and LOW priority header color prefix templates vary only in that HIGH sets 7 as a flag.
106 # By including 7 as a formatting flag (the order of flags does not matter) the text will trade
107 # foreground and background colors.
108 # E.G. for a critical message header, supposing the background
109 # is set to black by default, we will see black text on a bright-red background.
110 #
111 # As added emphasis, the order of flags is irrelevant. "1;4;7;Xm"<=>"7;1;4;Xm" or any other
112 # possible permutation of the ordering.

115 # https://en.wikipedia.org/wiki/ANSI_escape_code

257 def _logger_setup(logger: logging.Logger,
258  header_label: str,
259  formatter: Formatter = None,
260  level: int = logging.INFO,
261  level_normalized: int = logging.INFO,
262  handler_delegate=logging.StreamHandler,
263  **handler_kwargs) ->logging.Logger:
264  """A convenience function for creating well named loggers with optional custom formatter.
265 
266  This function will implement an already defined formatter for you if the formatter param is
267  None. For an example of a good general use logger see the example code at the bottom of this
268  file.
269 
270  SPECIAL NOTE:
271  Although this function's signature allows the caller to pass virtually any logging.Handler
272  subclass into the handler_delegate parameter, I've only tested this functionality against
273  the logging.StreamHandler class. If you wish to use it for others you will likely encounter
274  bugs. But if you are up to the task it seems like it could potentially be a helpful tool for
275  allowing the instantiation of a wide array of utility loggers under a single interface.
276 
277  :param header_label:
278  :param logger: an initialized logger object that needs to have its formatter and optinoal handlers set.
279 
280  :param child_name: The auxiliary logger you wish to create to handle some specific task. The
281  logger which maps to this child will set its level, handlers, and formatters
282  according to your inputs, allowing you to specify many different loggers to
283  manage _data output in a form that suites you.
284  :type child_name: A string
285 
286  :param formatter: An optional parameter that specifies the manner in which you wish to format
287  the available logging-event-_data as well as how you wish to present the
288  message _data for your log events. The default formatter will take one of
289  two styles, based the given `level`.
290 
291  For level==logging.INFO (20) and bellow:
292  log messages will be presented in two parts.
293  * First, a header line that's formatted to be bold, and underlined,
294  that gives the time-stamp for when the log was submitted, the
295  child_name, the model and function and line number from which the
296  log-message was originated
297  * Followed by an indented new-line where the log-message will be
298  be printed. The message
299  For level==logging.INFO+1 (21) and above:
300  log messages will be presented in two parts.
301  * First, a header line that's formatted to be bold, and underlined,
302  that gives the time-stamp for when the log was submitted, the
303  child_name, the model and function and line number from which the
304  log-message was originated
305  * Followed, on the same line as the header, the log-message. This
306  distinction, as opposed to the indented new-line in lower level
307  messaged, is done because it is often the case the when higher
308  level messages occur, there are very many of them. Forcing each
309  one to then be a multi-line message actually makes it much harder
310  to visually parse.
311  * Special note: In order to aid in automated parsing of these
312  log-messages, the header details and log message will be seperated
313  by the following character key:
314  `::>`
315 
316  :type formatter: an instance of logging.Formatter
317 
318  :param handler_delegate: An optional parameter that Specifies the type of handler you want to
319  associate to the logger instance that's mapped to the
320  root_name.child_name you've passed in. The handler will be set up
321  inside of this function, this parameter simply allows you to indicate
322  the way you wish to have your output handled.
323  (E.G. to std_out, std_err, or some file output stream)
324  :type handler_delegate: This should be a delegate function of your desired handler's
325  constructor, DEFAULT=logging.StreamHandler
326 
327  :param level: Specifies the desired logging level of the resulting logger.
328  DEFAULT=logging.DEBUG
329  :type level: An int, must be in the range of [0,0xffffffff]
330 
331  :type handler_kwargs: Any additional keywords that should be passed into the construction of the
332  handler. These are necessary for the instantiation of handlers that will
333  output to anything other than sys.std_out, and sys.std_err.
334 
335  :return: a reference to the logger instance that's mapped to the input naming scheme of
336  <root_name>.<child_name >
337  :rtype: logging.Logger
338  """
339  logger.propagate = False
340  level_normalized = level if level_normalized is None else level_normalized
341  try:
342  colr_id = log_id_map[level_normalized]
343  except KeyError:
344  # level_normalized isn't in our custom log_id_map
345  if level_normalized in logging._levelToName:
346  # but it has been registered with the logging library
347  LEVELS_STR2INT[logging._levelToName[level_normalized]] = level_normalized
348  for lvl,color_code in sorted(log_id_map.items(),key=lambda tpl:tpl[0]):
349  if lvl<=level_normalized:
350  colr_id = color_code
351  else:
352  break
353  else:
354  colr_id = log_id_map["NOTSET"]
355  log_id_map[level_normalized] = colr_id
356  if formatter is None:
357  formatter = _build_default_formatter(level,header_label,colr_id,formatter)
358  logger.addHandler(_configure_handler(formatter,level,handler_delegate,handler_kwargs))
359  return logger
