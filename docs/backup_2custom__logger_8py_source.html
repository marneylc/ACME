105 # HIGH, and LOW priority header color prefix templates vary only in that HIGH sets 7 as a flag.
106 # By including 7 as a formatting flag (the order of flags does not matter) the text will trade
107 # foreground and background colors.
108 # E.G. for a critical message header, supposing the background
109 # is set to black by default, we will see black text on a bright-red background.
110 #
111 # As added emphasis, the order of flags is irrelevant. "1;4;7;Xm"<=>"7;1;4;Xm" or any other
112 # possible permutation of the ordering.

260 def _logger_setup(logger: logging.Logger,
261  header_label: str,
262  formatter: Formatter = None,
263  level: int = logging.INFO,
264  level_normalized: int = logging.INFO,
265  handler_delegate=logging.StreamHandler,
266  **handler_kwargs) ->logging.Logger:
267  """A convenience function for creating well named loggers with optional custom formatter.
268 
269  This function will implement an already defined formatter for you if the formatter param is
270  None. For an example of a good general use logger see the example code at the bottom of this
271  file.
272 
273  SPECIAL NOTE:
274  Although this function's signature allows the caller to pass virtually any logging.Handler
275  subclass into the handler_delegate parameter, I've only tested this functionality against
276  the logging.StreamHandler class. If you wish to use it for others you will likely encounter
277  bugs. But if you are up to the task it seems like it could potentially be a helpful tool for
278  allowing the instantiation of a wide array of utility loggers under a single interface.
279 
280  :param header_label:
281  :param logger: an initialized logger object that needs to have its formatter and optinoal handlers set.
282 
283  :param child_name: The auxiliary logger you wish to create to handle some specific task. The
284  logger which maps to this child will set its level, handlers, and formatters
285  according to your inputs, allowing you to specify many different loggers to
286  manage _data output in a form that suites you.
287  :type child_name: A string
288 
289  :param formatter: An optional parameter that specifies the manner in which you wish to format
290  the available logging-event-_data as well as how you wish to present the
291  message _data for your log events. The default formatter will take one of
292  two styles, based the given `level`.
293 
294  For level==logging.INFO (20) and bellow:
295  log messages will be presented in two parts.
296  * First, a header line that's formatted to be bold, and underlined,
297  that gives the time-stamp for when the log was submitted, the
298  child_name, the model and function and line number from which the
299  log-message was originated
300  * Followed by an indented new-line where the log-message will be
301  be printed. The message
302  For level==logging.INFO+1 (21) and above:
303  log messages will be presented in two parts.
304  * First, a header line that's formatted to be bold, and underlined,
305  that gives the time-stamp for when the log was submitted, the
306  child_name, the model and function and line number from which the
307  log-message was originated
308  * Followed, on the same line as the header, the log-message. This
309  distinction, as opposed to the indented new-line in lower level
310  messaged, is done because it is often the case the when higher
311  level messages occur, there are very many of them. Forcing each
312  one to then be a multi-line message actually makes it much harder
313  to visually parse.
314  * Special note: In order to aid in automated parsing of these
315  log-messages, the header details and log message will be seperated
316  by the following character key:
317  `::>`
318 
319  :type formatter: an instance of logging.Formatter
320 
321  :param handler_delegate: An optional parameter that Specifies the type of handler you want to
322  associate to the logger instance that's mapped to the
323  root_name.child_name you've passed in. The handler will be set up
324  inside of this function, this parameter simply allows you to indicate
325  the way you wish to have your output handled.
326  (E.G. to std_out, std_err, or some file output stream)
327  :type handler_delegate: This should be a delegate function of your desired handler's
328  constructor, DEFAULT=logging.StreamHandler
329 
330  :param level: Specifies the desired logging level of the resulting logger.
331  DEFAULT=logging.DEBUG
332  :type level: An int, must be in the range of [0,0xffffffff]
333 
334  :type handler_kwargs: Any additional keywords that should be passed into the construction of the
335  handler. These are necessary for the instantiation of handlers that will
336  output to anything other than sys.std_out, and sys.std_err.
337 
338  :return: a reference to the logger instance that's mapped to the input naming scheme of
339  <root_name>.<child_name >
340  :rtype: logging.Logger
341  """
342  logger.propagate = False
343  level_normalized = level if level_normalized is None else level_normalized
344  try:
345  colr_id = log_id_map[level_normalized]
346  except KeyError:
347  # level_normalized isn't in our custom log_id_map
348  if level_normalized in logging._levelToName:
349  # but it has been registered with the logging library
350  LEVELS_STR2INT[logging._levelToName[level_normalized]] = level_normalized
351  for lvl,color_code in sorted(log_id_map.items(),key=lambda tpl:tpl[0]):
352  if lvl<=level_normalized:
353  colr_id = color_code
354  else:
355  break
356  else:
357  colr_id = log_id_map["NOTSET"]
358  log_id_map[level_normalized] = colr_id
359  if formatter is None:
360  formatter = _build_default_formatter(level,header_label,colr_id,formatter)
361  logger.addHandler(_configure_handler(formatter,level,handler_delegate,handler_kwargs))
362  return logger
