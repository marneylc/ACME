123 def all_to_intermediary(filename_or_input, schema=None):
124  """ Dispatch the filename_or_input to the different function to produce the intermediary syntax.

126  The input can also be a list of strings in markdown format or a filename finishing by '.er' containing markdown

128  """

177 def filter_resources(tables, relationships,
178  include_tables=None, include_columns=None,
179  exclude_tables=None, exclude_columns=None):
180  """
181  Include the following:
182  1. Tables and relationships with tables present in the include_tables (lst of str, tables names)
183  2. Columns (of whichever table) present in the include_columns (lst of str, columns names)
184  Exclude the following:
185  1. Tables and relationships with tables present in the exclude_tables (lst of str, tables names)
186  2. Columns (of whichever table) present in the exclude_columns (lst of str, columns names)
187  Disclosure note:
188  All relationships are taken into consideration before ignoring columns.
189  In other words, if one excludes primary or foreign keys, it will still keep the relations display amongst tables
190  """
191  def convert_to_set(target,default_data=None):
192  if target is not None:
193  if not isinstance(target, set):
194  target = set(target)
195  else:
196  if default_data is None:
197  target = set()
198  else:
199  target = set(default_data)
200  return target
201 
202  _tables = copy.deepcopy(tables)
203  _relationships = copy.deepcopy(relationships)
204 
205  exclude_tables = convert_to_set(exclude_tables)
206  include_tables = convert_to_set(include_tables, (t.name for t in _tables if t.name not in exclude_tables))
207  exclude_columns = convert_to_set(exclude_columns)
208  include_columns = convert_to_set(include_columns, (c.name for t in _tables for c in t.columns if c.name not in exclude_columns))
209 
210  _tables = [t for t in _tables
211  if t.name in include_tables]
212  _relationships = [r for r in _relationships
213  if r.right_col in include_tables and r.left_col in include_tables]
214 
215  for t in _tables:
216  t.columns = [c for c in t.columns if c.name in include_columns]
217 
218  return _tables, _relationships
